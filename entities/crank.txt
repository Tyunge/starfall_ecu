dofile('engine_ecu/entities/entity.txt')

Crank = class('Crank',Entity)
function Crank:initialize()
    self.mass = 0
    self.inertia = Vector(5,5,3)
    self.entity = nil
    self.last_tick_rpm = 0
    self.sound = nil
end

function Crank:getRPM()
    return math.abs(self.entity:getAngleVelocity():dot(Vector(0,0,1))/6)
end

function Crank:getDelta()
    local rpm = self:getRPM()
    local delta = rpm-self.last_tick_rpm
    self.last_tick_rpm = self:getRPM()
    return delta
end

function Crank:setSound(path)
    self.sound = sounds.create(self.entity,path)
        self.sound:play()
end

function Crank:getSound()
    return self.sound
end
